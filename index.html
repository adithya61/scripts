Dislikes Script
const url = "https://leetcode.com/graphql/"; const siteUrl = window.location.href; const slug = siteUrl.split("/")[4]; let variables = { titleSlug: slug }; const query = "\n query questionTitle($titleSlug: String!) {question(titleSlug: $titleSlug) {likes\n dislikes}}"; const options = { method: "POST", headers: { "Content-Type": "application/json", }, body: JSON.stringify({ query: query, variables: variables, }), }; const res = await fetch(url, options); let favCount = await res.json().then((data) => data["data"]["question"]); console.log(favCount); // {likes: ..., dislikes: ...}; let v = document.getElementsByClassName( `relative inline-flex gap-2 items-center justify-center font-medium cursor-pointer focus-visible:outline-none disabled:cursor-not-allowed disabled:opacity-50 transition-colors bg-transparent enabled:hover:bg-fill-secondary enabled:active:bg-fill-primary text-body rounded-none p-2 text-text-secondary dark:text-text-secondary hover:text-text-secondary dark:hover:text-text-secondary` )[0].innerHTML += favCount['dislikes'];
